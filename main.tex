	\frametitle{What are Stratigraphic Intervals?} \pause
	\begin{itemize}
		\item The range in which a species/lineage exists in time; \pause
		\item Can be measured in time or column depth; \pause
		\item Useful for: \pause
		      \begin{itemize}
			      \item Calibrating phylogenies; \pause
			      \item Calculating evolutionary rates;
		      \end{itemize}
	\end{itemize}

\begin{frame}
	\frametitle{Using Bayesian inference to estimate endpoints} \pause

	Bayes Theorem:
	\begin{equation}
		P(\theta\cdot|D) = \frac{P(\theta\cdot)P(D|\theta\cdot)}{\int_{\Theta}P(\theta)P(D|\theta)d\theta}
		\label{eq:bayes-theorem}
	\end{equation} \pause

	\begin{itemize}
		\item $P(\theta\cdot)$: prior distribution \pause
		\item $P(D|\theta\cdot)$: likelihood function \pause
		\item $P(\theta\cdot|D)$: posterior distribution \pause
	\end{itemize}

	Parameters of interest: $\theta_1$ and $\theta_2$; \pause

	We need some likelihood function that encodes endpoints as parameters.
\end{frame}

\begin{frame}
	\frametitle{Objectives of this work} \pause

	Test the inference power of the four-parameter Beta distribution in endpoint estimation \pause

	\begin{itemize}
		\item Does it converge as easily as the three-parameter variant? \pause
		\item Can it estimate endpoints correctly in a wider range of scenarios? \pause
		\item How priors influence the posterior distribution?
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Tests with simulated datasets} \pause

	\begin{itemize}
		\item $\theta_1 =$ 150 MA, $\theta_2 =$ 100 MA; \pause
		\item $\alpha$ and $\beta$ varying between 1, 2, and 10; \pause
		\item Sample size varying between 10, 50, 100, 200 and 500 samples. \pause
	\end{itemize}

	We tested: \pause
	\begin{itemize}
		\item Correct and incorrect ($\mu$), informative and uninformative ($\sigma$) Normal priors; \pause
		\item Measures of mean, inter-quartile range (IQR) and mean squared errors (MSE) of posteriors;
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Empirical example: superorder Ostariophysi} \pause

	Who are the ostariophysins? \pause

	Superorder of freshwater bony fish \pause

	\begin{itemize}
		\item Characiformes (pacus and piranhas);
		\item Siluriformes (catfishes and plecos);
		\item Gymnotiformes (electric fish);
		\item Cypriniformes (carps);
		\item Gonorynchiformes (milkfish);
	\end{itemize} \pause

	Also part of the greater clade Otocephala, which includes:
	\begin{itemize}
		\item Clupeomorpha (anchovies);
		\item Alepocephali (nakedheads);
	\end{itemize} \pause

\end{frame}

\begin{frame}
	\frametitle{What we tested} \pause

	\begin{itemize}
		\item The origins of each order (except Gymnotiformes); \pause
		\item The origins of Ostariophysi and Otocephala as a whole; \pause
    \item Exponential priors; \pause
		\item Parameter $\theta_2$ (time of extinction) always fixed in the present; \pause
		\item Filtering out data with same time range and locality; \pause
	\end{itemize}

	All data was sampled from the PaleoBiology DataBase (ref);
\end{frame}

\begin{frame}
	\frametitle{Results} \pause

	\begin{itemize}
		\item All analyses converged (ESS > 1000;); \pause
		      \begin{itemize}
			      \item Except for Gonorynchiformes when using four parameters (ESS of tail > 400); \pause
		      \end{itemize}
		\item Posterior probabilities (PP) indicate the four-parameter model as the best to fit the data with $\approx100\%$ probability; \pause
		      \begin{itemize}
			      \item Except for two cases: Characiformes and Gonorynchiformes; \pause
		      \end{itemize}
		\item Almost all groups with origins predicted in the Late Jurassic-Early Cretaceous;
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Conclusions} \pause

	\begin{itemize}
		\item The four parameter distribution is more versatile and models a wider range of fossil distributions; \pause
		\item It also shows convergence similar to that of three parameters, even when the latter could be used instead; \pause
		\item Wider, less informative priors are better for inference with low impact on correctness.
	\end{itemize}
