\begin{frame}
	\frametitle{Using Bayesian inference to estimate endpoints} \pause

	Bayes Theorem:
	\begin{equation}
		P(\theta\cdot|D) = \frac{P(\theta\cdot)P(D|\theta\cdot)}{\int_{\Theta}P(\theta)P(D|\theta)d\theta}
		\label{eq:bayes-theorem}
	\end{equation} \pause

	\begin{itemize}
		\item $P(\theta\cdot)$: prior distribution \pause
		\item $P(D|\theta\cdot)$: likelihood function \pause
		\item $P(\theta\cdot|D)$: posterior distribution \pause
	\end{itemize}

	Parameters of interest: $\theta_1$ and $\theta_2$; \pause

	We need some likelihood function that encodes endpoints as parameters.
\end{frame}

\begin{frame}
	\frametitle{The StratIntervals.jl package} \pause

	Likelihood: Three-parameter Beta distribution (ref.)
	\begin{equation}
		f(\tau|\lambda,\theta_1,\theta_2) =
		\begin{cases}
			\frac{(\theta_1 - \tau)^{-\lambda}}{(\theta_1 - \theta_2)^{1-\lambda}B(1, 1-\lambda)} & \text{if} \lambda \leq 0 \\
			\frac{(\tau - \theta_2)^{\lambda}}{(\theta_1 - \theta_2)^{1+\lambda}B(1+\lambda, 1)}  & \text{if} \lambda > 0
		\end{cases}
		\label{eq:three-par-beta}
	\end{equation} \pause

	Usage of MCMC algorithms provided by Turing (ref.); \pause

	\emph{*insert GIF of three-par beta graph examples*}
\end{frame}

\begin{frame}
	\frametitle{Using four parameters}
	\begin{equation}
		f(\tau|\lambda,\theta_1,\theta_2) = \frac{(\theta_1 - \tau)^{\alpha-\beta}}{(\theta_1 - \theta_2)^{\alpha+\beta-1}B(\alpha, \beta)}
	\end{equation}

	More parameters = harder to converge \pause

	But greater variety of curves!

	\emph{*insert GIF of four-par beta graph examples*}
\end{frame}

\begin{frame}
	\frametitle{Objectives of this work} \pause

	Test the inference power of the four-parameter Beta distribution in endpoint estimation \pause

	\begin{itemize}
		\item Does it converge as easily as the three-parameter variant? \pause
		\item Can it estimate endpoints correctly in a wider range of scenarios? \pause
		\item How priors influence the posterior distribution?
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Experimental framework} \pause

	\textbf{Simulations:} \pause
	\begin{itemize}
		\item $\theta_1 =$ 150 MA, $\theta_2 =$ 100 MA; \pause
		\item $\alpha$ and $\beta$ varying between 1, 2, and 10; \pause
		\item Sample size varying between 10, 50, 100, 200 and 500 samples. \pause

	We tested: \pause
	\begin{itemize}
		\item Sampling from only one or both endpoints; \pause
		\item Correct and incorrect ($\mu$), informative and uninformative ($\sigma$) priors; \pause
		\item Measures of mean, inter-quartile range (IQR) and mean squared errors (MSE) of posteriors;
	\end{itemize}
